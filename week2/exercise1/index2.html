Exercise 1, Part 2

Repeat a given string (first argument) num times (second argument). Return an empty string if num is not a positive number.

We solve it using a for loop

    function repeatStringNumTimes(str, num) {
      // repeat after me
      if (num > 0) {
        var repeatStr = str;
        for (var i = 1; i < num; i++) {
          str = str + repeatStr;
        }
        return str;
      } else {
        return "";
      }
    }

    repeatStringNumTimes("abc", 3);

Now we solve it using a while loop

    function repeatStringNumTimes(str, num) {
      // repeat after me
      if (num > 0) {
        var repeatStr = str;
        var i = 1;
        while (i < num) {
          str = str + repeatStr;
          i++;
        }
        return str;
      } else {
        return "";
      }
    }

    repeatStringNumTimes("abc", 3);

And with a do-while loop

    function repeatStringNumTimes(str, num) {
      // repeat after me
      if (num > 0) {
        var repeatStr = str;
        var i = 0;
        str = "";
        do {
          str = str + repeatStr;
          i++;
        } while (i < num);
        return str;
      } else {
        return "";
      }
    }
    
    repeatStringNumTimes("abc", 3);
